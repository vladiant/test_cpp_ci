# Catch2 (catch.hpp) is a header-only library compiled into the test binary.
# It makes heavy use of std::string, std::map and other libstdc++ types whose
# backing memory is allocated by uninstrumented libstdc++, producing a long
# tail of false-positive "use-of-uninitialized-value" reports from various
# Catch2 internal functions (ReporterRegistrar, ConsoleReporter, etc.).
# Suppress all warnings that originate from within the catch.hpp header.
src:*/catch2/catch.hpp
